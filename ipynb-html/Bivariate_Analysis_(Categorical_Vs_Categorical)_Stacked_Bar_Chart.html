# Create the contingency table for Embarked vs Survived
survived_embarked = pd.crosstab(titanic['embarked'], titanic['survived'])

# Create the stacked bar chart
plt.figure(figsize=(8, 6))
survived_embarked.plot(kind='bar', stacked=True, color=["lightgreen", "lightcoral"])
plt.title("Embarked vs Survived")
plt.xlabel("Embarked")
plt.ylabel("Count")
plt.xticks(rotation=0)
plt.legend(title="Survived", labels=['No', 'Yes'])

# Add value labels to the bars inside the segments
for i, (index, row) in enumerate(survived_embarked.iterrows()):
    not_survived = row[0]
    survived = row[1]

    # Label for 'Not Survived' segment (bottom)
    plt.text(i, not_survived / 2, f'{int(not_survived)}', ha='center', va='center', color='black', fontsize=9)

    # Label for 'Survived' segment (top)
    plt.text(i, not_survived + survived / 2, f'{int(survived)}', ha='center', va='center', color='black', fontsize=9)

plt.tight_layout()
plt.show();

# --- Stacked Bar Charts ---

variables_to_plot = [
    ('Embarked', survived_embarked),
    ('Pclass', survived_pclass),
    ('Sex', survived_sex),
    ('Class', survived_class),
    ('Adult Male', survived_adult_male),
    ('Embark Town', survived_embark_town),
    ('Alone', survived_alone)
]

color_choices = [
    ["skyblue", "salmon"],
    ["skyblue", "salmon"],  # Example color pairs
    ["lightgreen", "lightcoral"],
    ["gold", "lightslategrey"],
    ["plum", "lightcyan"],
    ["wheat", "thistle"],
    ["lightcoral", "lightblue"]
]

num_plots = len(variables_to_plot)
rows = (num_plots + 2) // 3  # Adjust rows based on number of plots
cols = min(num_plots, 3)

fig, axes = plt.subplots(rows, cols, figsize=(18, 7 * rows))
fig.suptitle("Stacked Bar Charts of Survival vs. Other Categorical Variables", fontsize=16, y=1.02)

axes = axes.flatten()  # Flatten the axes array for easy indexing

for i, (var_name, cont_table) in enumerate(variables_to_plot):
    if i < len(axes):
        colors = color_choices[i % len(color_choices)]  # Cycle through color choices
        cont_table.T.plot(kind='bar', stacked=True, ax=axes[i], color=colors)
        axes[i].set_title(f"Survived vs {var_name}")
        axes[i].set_xlabel(var_name)
        axes[i].set_ylabel("Count")
        # Adjust tick label alignment using set_ticklabels
        axes[i].set_xticklabels(axes[i].get_xticklabels(), rotation=45, ha='left')
        axes[i].legend(title="Survived", loc='upper left')

# Remove any unused subplots
if num_plots < len(axes):
    for j in range(num_plots, len(axes)):
        fig.delaxes(axes[j])

plt.tight_layout(rect=[0, 0.03, 1, 0.95])
plt.show()

# --- Stacked Bar Charts with Value Labels ---

variables_to_plot = [
    ('Embarked', survived_embarked),
    ('Pclass', survived_pclass),
    ('Sex', survived_sex),
    ('Class', survived_class),
    ('Adult Male', survived_adult_male),
    ('Embark Town', survived_embark_town),
    ('Alone', survived_alone)
]

color_choices = [
    ["skyblue", "salmon"],
    ["skyblue", "salmon"],  # Example color pairs
    ["lightgreen", "lightcoral"],
    ["gold", "lightslategrey"],
    ["plum", "lightcyan"],
    ["wheat", "thistle"],
    ["lightcoral", "lightblue"]
]

num_plots = len(variables_to_plot)
rows = (num_plots + 2) // 3
cols = min(num_plots, 3)

fig, axes = plt.subplots(rows, cols, figsize=(18, 7 * rows))
fig.suptitle("Stacked Bar Charts of Survival vs. Other Categorical Variables with Values", fontsize=16, y=1.02)

axes = axes.flatten()

for i, (var_name, cont_table) in enumerate(variables_to_plot):
    if i < len(axes):
        colors = color_choices[i % len(color_choices)]
        ax = axes[i]
        cont_table.T.plot(kind='bar', stacked=True, ax=ax, color=colors)
        ax.set_title(f"Survived vs {var_name}")
        ax.set_xlabel(var_name)
        ax.set_ylabel("Count")
        # Changed line to fix the error
        ax.tick_params(axis='x', labelrotation=45)  # Use labelrotation for x-axis label rotation
        ax.legend(title="Survived", loc='upper left')

        # Add value labels to the bars
        for p in ax.patches:
            width = p.get_width()
            height = p.get_height()
            x, y = p.get_xy()
            if height > 0:  # Only add label if the height is not zero
                ax.annotate(f'{int(height)}', (x + width/2., y + height/2.),
                            ha='center', va='center', fontsize=9)

# Remove any unused subplots
if num_plots < len(axes):
    for j in range(num_plots, len(axes)):
        fig.delaxes(axes[j])

plt.tight_layout(rect=[0, 0.03, 1, 0.95])
plt.show()

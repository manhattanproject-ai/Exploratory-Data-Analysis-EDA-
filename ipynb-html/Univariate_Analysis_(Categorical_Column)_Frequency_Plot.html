# Get the value counts of the 'Embarked' column
embarked_counts = df['embarked'].value_counts().reset_index()
embarked_counts.columns = ['Embarked', 'Frequency']  # Rename for clarity

# Calculate percentages for the frequency table
embarked_counts['Percentage'] = (embarked_counts['Frequency'] / len(df)) * 100

# Print the frequency table
print(embarked_counts)

# Create a bar plot
plt.figure(figsize=(8, 6))  # Adjust figure size as needed
sns.barplot(x='Embarked', y='Frequency', data=embarked_counts)

# Add a title and labels
plt.title('Embarked Frequency Plot')
plt.xlabel('Port of Embarkation')
plt.ylabel('Frequency')

# Add percentage labels on the bars
for index, row in embarked_counts.iterrows():
    plt.text(row.Embarked, row.Frequency, f'{row.Percentage:.1f}%', ha='center', va='bottom')

# Show the plot
plt.show()

# Define the columns to plot
columns_to_plot = ['survived', 'pclass', 'sex', 'sibsp', 'parch', 'embarked', 'class', 'who', 'embark_town', 'alive', 'alone']

# Define a list of color palettes
color_palettes = ['Blues', 'Greens', 'Reds', 'Purples', 'Oranges', 'YlGnBu', 'YlGn', 'PuBu', 'PuBuGn', 'BuPu', 'GnBu']

# Determine the number of rows and columns for the subplot grid
num_cols = 4
num_rows = (len(columns_to_plot) + num_cols - 1) // num_cols  # Ensure enough rows

# Create the subplots
fig, axes = plt.subplots(num_rows, num_cols, figsize=(18, 12))
axes = axes.flatten()

# Iterate through the columns and plot each one in a separate subplot
for i, col in enumerate(columns_to_plot):
    # Select the appropriate subplot axis
    ax = axes[i]

    # Use a different color palette for each bar plot
    palette = color_palettes[i % len(color_palettes)]  # Cycle through palettes

    # Create the bar plot using countplot with the selected palette
    sns.countplot(x=col, data=df, ax=ax, palette=palette)

    # Add a title and labels
    ax.set_title(f"Frequency Plot of {col}")
    ax.set_xlabel(col.title())
    ax.set_ylabel("Frequency")

    # Add percentage labels on the bars
    total = len(df[col])
    for p in ax.patches:
        height = p.get_height()
        ax.text(p.get_x() + p.get_width() / 2., height + 3, f'{100 * height / total:.1f}%', ha='center', va='bottom')

# Remove any unused subplots
if len(columns_to_plot) < num_rows * num_cols:
    for j in range(len(columns_to_plot), num_rows * num_cols):
        fig.delaxes(axes[j])

# Adjust the layout to prevent overlapping titles/labels
plt.tight_layout()

# Show the plots
plt.show()

4. How to Interpret Frequency plot :

Key Features to Look For :

  • Bars: The plot consists of bars, each representing a category.
  • Bar Height: The height of each bar corresponds to the frequency (count) of observations in that category. Taller bars indicate higher frequencies.
  • Categories: The x-axis displays the categories being analyzed.
  • Frequency: The y-axis represents the frequency (count) of observations for each category.
  • Colors: Different colors may be used to distinguish between bars or highlight specific categories.

Frequency Plot of Survived

  • Categories: 0 (Did not survive), 1 (Survived)
  • Bar Heights: The bar for '0' is taller than the bar for '1'.
  • Frequency: The y-axis shows the number of passengers.
  • Observations: A greater number of passengers did not survive compared to those who did.

Frequency Plot of Pclass

  • Categories: 1, 2, 3 (Passenger Class)
  • Bar Heights: The bar for '3' is the tallest, followed by '1', and then '2'.
  • Frequency: The y-axis shows the number of passengers.
  • Observations: Most passengers were in the third class, followed by the first class, and the fewest in the second class.

Frequency Plot of Sex

  • Categories: male, female
  • Bar Heights: The bar for 'male' is taller than the bar for 'female'.
  • Frequency: The y-axis shows the number of passengers.
  • Observations: There were more male passengers than female passengers.

Frequency Plot of Sibsp

  • Categories: 0, 1, 2, 3, 4, 5, 8 (Number of Siblings/Spouses)
  • Bar Heights: The bar for '0' is the tallest, with a decreasing height as the number of siblings/spouses increases.
  • Frequency: The y-axis shows the number of passengers.
  • Observations: Most passengers had no siblings or spouses, and the number of passengers decreases as the number of siblings/spouses increases.

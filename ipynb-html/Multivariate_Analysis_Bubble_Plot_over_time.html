# Convert 'Year' to categorical for proper spacing on x-axis

df['Year'] = df['Year'].astype('category')

plt.figure(figsize=(12, 8))
sns.scatterplot(data=df, x='Year', y='Country', size='Exports', hue='Exports',
                sizes=(20, 500), palette='coolwarm', legend='full')

plt.title('Exports by Country Over Time')
plt.xlabel('Year')
plt.ylabel('Country')

# Adjust legend position

plt.grid(axis='x', linestyle='--')
plt.tight_layout()
